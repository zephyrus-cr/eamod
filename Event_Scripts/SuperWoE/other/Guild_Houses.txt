function	script	rent_pvp	{
	if (getmapxy(@mapa$,@mapx,@mapy,0,strcharinfo(0)) != 0) end;
	mes "[Switch]";
	mes "What do you want to do?";
	next;
	menu "Enable PVP",L_PvpOn,"Disable PVP",L_PvpOff,"Nothing",-;
	mes "[Switch]";
	mes "Good Day";
	close;

	mes "[Switch]";
	mes "PVP Mode Enable";
	pvpon @mapa$;
	close;

L_PvpOff:
	mes "[Switch]";
	mes "PVP Mode Disable";
	pvpoff @mapa$;
	close;
}

//============================================================ 
//= Funciones de Inicio y Fin de Reproduccin de Piano
//= (id de Casa)
//============================================================ 

function	script	rent_pianoStart	{
	if (getmapxy(@mapa$,@mapx,@mapy,0,strcharinfo(0)) != 0) end;
	mes "[Piano]";
	if (getd("$@Piano" + getarg(0)) > 0) {
		mes "Please wait for the piano to be ready again.";
		close;
	}
	mes "What do you want to play?";
	set @Tema, select ("Chaos of Eternity","Bragis Poem","Ring of Nibelungen","Dont Forget Me Not","In To The Abyss","Assassin of Sunset","Ninguno");
	if (@Tema >= 7) close;
	if (@Tema < 10) set @Tema$, "m0" + @Tema + ".wav";
	else set @Tema$, "m" + @Tema + ".wav";
	close2;
	if (getd("$@Piano" + getarg(0)) <= 0) {
		// Sonar el Tema Elegido
		setd "$@Piano" + getarg(0), 1; // Reproduciendo
		initnpctimer;
		switch (@Tema) {
			case 1:  soundeffectall "chaos_of_eternity.wav",0,@mapa$,0,0,50,50; break;
			case 2:  soundeffectall "bragis_poem.wav",0,@mapa$,0,0,50,50; break;
			case 3:  soundeffectall "ring_of_nibelungen.wav",0,@mapa$,0,0,50,50; break;
			case 4:  soundeffectall "dont_forget_me_not.wav",0,@mapa$,0,0,50,50; break;
			case 5:  soundeffectall "in_to_the_abyss.wav",0,@mapa$,0,0,50,50; break;
			case 6:  soundeffectall "assassin_of_sunset.wav",0,@mapa$,0,0,50,50; break;
		}
	}
	end;
}

function	script	rent_pianoStop	{
	emotion e_no1;
	stopnpctimer;
	setd "$@Piano" + getarg(0), 0; // Fin de Tema
	end;
}
